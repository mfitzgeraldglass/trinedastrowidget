<!DOCTYPE html>

<html>

<head>

    <title>Cancer Horoscope Widget</title>

    <style>

        body {

            font-family: Arial, sans-serif;

            background-color: #000080; /* Dark Blue */

            color: #FFFFFF; /* White */

        }

        h1 {

            color: #0000CD; /* Medium Blue */

        }

        p {

            color: #D3D3D3; /* Light Gray */

        }

        #astrology {

            background-color: #ADD8E6; /* Light Blue */

            padding: 20px;

            border-radius: 10px;

        }

    </style>

</head>

<body>

    <div id="astrology">

        <!-- The horoscope data will go here -->

    </div>


    <script>

        // This is the URL for the Aztro API for the sign Cancer

        const url = 'https://aztro.sameerkumar.website/?sign=cancer&day=today';


        // Fetch data from the Aztro API

        fetch(url, {

            method: 'POST'

        })

        .then(response => response.json())

        .then(data => {

            // Once we have the data, update the 'astrology' div with the horoscope data

            document.getElementById('astrology').innerHTML = `

                <h1>Today's Horoscope for Cancer</h1>

                <p><strong>Mood:</strong> ${data.mood}</p>

                <p><strong>Lucky Number:</strong> ${data.lucky_number}</p>

                <p><strong>Lucky Time:</strong> ${data.lucky_time}</p>

                <p><strong>Color:</strong> ${data.color}</p>

                <p><strong>Description:</strong> ${data.description}</p>

            `;

        });

    </script>

</body>
